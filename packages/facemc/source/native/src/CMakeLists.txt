# Find all headers and sources (no updates to this file necessary!)
FILE(GLOB FACEMC_SOURCE_NATIVE_HEADERS *.hpp)
FILE(GLOB FACEMC_SOURCE_NATIVE_SOURCES *.cpp)

IF(CMAKE_VERBOSE_CONFIGURE)
  MESSAGE("FACEMC_SOURCE_NATIVE_HEADERS = ${FACEMC_SOURCE_NATIVE_HEADERS}")
  MESSAGE("FACEMC_SOURCE_NATIVE_SOURCES = ${FACEMC_SOURCE_NATIVE_SOURCES}")
ENDIF()

SET(SUBPACKAGE_LIB_NAME facemc_source_native)

# Create the facemc_source_native library
ADD_LIBRARY(${SUBPACKAGE_LIB_NAME} ${FACEMC_SOURCE_NATIVE_SOURCES})
TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${TEUCHOS_CORE} utility_prng utility_dist facemc_core)
IF(FRENSIE_ENABLE_DAGMC)
  TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} geometry_dagmc)
ENDIF()
IF(FRENSIE_ENABLE_MPI)
  TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${MPI_CXX_LIBRARIES})
ENDIF()

INSTALL(TARGETS ${SUBPACKAGE_LIB_NAME}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/lib)
INSTALL(FILES ${FACEMC_SOURCE_NATIVE_HEADERS}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/include
  PERMISSIONS OWNER_READ GROUP_READ WORLD_READ)

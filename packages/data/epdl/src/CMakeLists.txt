# Find all headers and sources (no updates to this file necessary!)
FILE(GLOB DATA_EPDL_HEADERS *.hpp)
FILE(GLOB DATA_EPDL_SOURCES *.cpp)

IF(CMAKE_VERBOSE_CONFIGURE)
  MESSAGE("DATA_EPDL_HEADERS = ${DATA_EPDL_HEADERS}")
  MESSAGE("DATA_EPDL_SOURCES = ${DATA_EPDL_SOURCES}")
ENDIF()

SET(SUBPACKAGE_LIB_NAME dataepdl)

# Create the utilitycore library
ADD_LIBRARY(${SUBPACKAGE_LIB_NAME} ${DATA_EPDL_SOURCES})
TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} utilitycore utilityhdf5)
IF(${FRENSIE_ENABLE_MPI})
  TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${MPI_CXX_LIBRARIES})
ENDIF()

INSTALL(TARGETS ${SUBPACKAGE_LIB_NAME}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/lib)
INSTALL(FILES ${DATA_EPDL_HEADERS}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/include
  PERMISSIONS OWNER_READ GROUP_READ WORLD_READ)

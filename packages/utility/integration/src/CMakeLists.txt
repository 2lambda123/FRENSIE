# Find all headers and sources (no updates to this file necessary!)
FILE(GLOB UTILITY_INTEGRATION_HEADERS *.hpp)
FILE(GLOB UTILITY_INTEGRATION_SOURCES *.cpp *.F90)

IF(CMAKE_VERBOSE_CONFIGURE)
  MESSAGE("UTILITY_INTEGRATION_HEADERS = ${UTILITY_INTEGRATION_HEADERS}")
  MESSAGE("UTILITY_INTEGRATION_SOURCES = ${UTILITY_INTEGRATION_SOURCES}")
ENDIF()

SET(SUBPACKAGE_LIB_NAME utility_integration)

# Create the utility integration library
ADD_LIBRARY(${SUBPACKAGE_LIB_NAME} ${UTILITY_INTEGRATION_SOURCES})
TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} utility_core)
IF(${FACEMC_ENABLE_MPI})
  TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${MPI_CXX_LIBRARIES})
ENDIF()

INSTALL(TARGETS ${SUBPACKAGE_LIB_NAME}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/lib)
INSTALL(FILES ${UTILITY_INTEGRATION_HEADERS}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/include
  PERMISSIONS OWNER_READ GROUP_READ WORLD_READ)

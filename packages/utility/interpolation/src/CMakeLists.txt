# Find all headers and sources (no updates to this file necessary!)
FILE(GLOB UTILITY_INTERPOLATION_HEADERS *.hpp)
FILE(GLOB UTILITY_INTERPOLATION_SOURCES *.cpp)

IF(CMAKE_VERBOSE_CONFIGURE)
  MESSAGE("UTILITY_INTERPOLATION_HEADERS = ${UTILITY_INTERPOLATION_HEADERS}")
  MESSAGE("UTILITY_INTERPOLATION_SOURCES = ${UTILITY_INTERPOLATION_SOURCES}")
ENDIF()

SET(SUBPACKAGE_LIB_NAME utility_interp)

# Create the utilitycore library
ADD_LIBRARY(${SUBPACKAGE_LIB_NAME} ${UTILITY_INTERPOLATION_SOURCES})
TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} utility_core)

INSTALL(TARGETS ${SUBPACKAGE_LIB_NAME}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/lib)
INSTALL(FILES ${UTILITY_INTERPOLATION_HEADERS}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/include
  PERMISSIONS OWNER_READ GROUP_READ WORLD_READ)

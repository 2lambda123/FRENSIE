# Find all headers and sources (no updates to this file necessary!)
FILE(GLOB FACEMC_CORE_HEADERS *.hpp)
FILE(GLOB FACEMC_CORE_SOURCES *.cpp)

IF(CMAKE_VERBOSE_CONFIGURE)
  MESSAGE("FACEMC_CORE_HEADERS = ${FACEMC_CORE_HEADERS}")
  MESSAGE("FACEMC_CORE_SOURCES = ${FACEMC_CORE_SOURCES}")
ENDIF()

SET(SUBPACKAGE_LIB_NAME facemc_core)

# Create the utilitycore library
ADD_LIBRARY(${SUBPACKAGE_LIB_NAME} ${FACEMC_CORE_SOURCES})
TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${TEUCHOS_CORE} utility_core utility_hdf5 geometry_core)
IF(${FACEMC_ENABLE_MPI})
  TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${MPI_CXX_LIBRARIES})
ENDIF()

INSTALL(TARGETS ${SUBPACKAGE_LIB_NAME}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/lib)
INSTALL(FILES ${FACEMC_CORE_HEADERS}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/include
  PERMISSIONS OWNER_READ GROUP_READ WORLD_READ)

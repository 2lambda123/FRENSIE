# Find all headers and sources (no updates to this file necessary!)
FILE(GLOB DATA_DECAY_HEADERS *.hpp)
FILE(GLOB DATA_DECAY_SOURCES *.cpp *.F90)

IF(CMAKE_VERBOSE_CONFIGURE)
  MESSAGE("DATA_DECAY_HEADERS = ${DATA_DECAY_HEADERS}")
  MESSAGE("DATA_DECAY_SOURCES = ${DATA_DECAY_SOURCES}")
ENDIF()

SET(SUBPACKAGE_LIB_NAME data_decay)

# Create the data_decay library
ADD_LIBRARY(${SUBPACKAGE_LIB_NAME} ${DATA_DECAY_SOURCES})
TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} utility_core ${Boost_LIBRARIES})
IF(${FRENSIE_ENABLE_MPI})
  TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${MPI_CXX_LIBRARIES})
ENDIF()

INSTALL(TARGETS ${SUBPACKAGE_LIB_NAME}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/lib)
INSTALL(FILES ${DATA_DECAY_HEADERS}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/include
  PERMISSIONS OWNER_READ GROUP_READ WORLD_READ)

# Find all headers and sources (no updates to this file necessary!)
FILE(GLOB DATA_ENDL_HEADERS *.hpp)
FILE(GLOB DATA_ENDL_SOURCES *.cpp *.F90)

IF(CMAKE_VERBOSE_CONFIGURE)
  MESSAGE("DATA_ENDL_HEADERS = ${DATA_ENDL_HEADERS}")
  MESSAGE("DATA_ENDL_SOURCES = ${DATA_ENDL_SOURCES}")
ENDIF()

SET(SUBPACKAGE_LIB_NAME data_endl)

# Create the utilitycore library
ADD_LIBRARY(${SUBPACKAGE_LIB_NAME} ${DATA_ENDL_SOURCES})
TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${Boost_LIBRARIES} data_core utility_core)
IF(${FRENSIE_ENABLE_MPI})
  TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${MPI_CXX_LIBRARIES})
ENDIF()

INSTALL(TARGETS ${SUBPACKAGE_LIB_NAME}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/lib)
INSTALL(FILES ${DATA_ENDL_HEADERS}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/include
  PERMISSIONS OWNER_READ GROUP_READ WORLD_READ)

# Find all headers and sources (no updates to this file necessary!)
FILE(GLOB FACEMC_ESTIMATOR_NATIVE_HEADERS *.hpp)
FILE(GLOB FACEMC_ESTIMATOR_NATIVE_SOURCES *.cpp)

IF(CMAKE_VERBOSE_CONFIGURE)
  MESSAGE("FACEMC_ESTIMATOR_NATIVE_HEADERS = ${FACEMC_ESTIMATOR_NATIVE_HEADERS}")
  MESSAGE("FACEMC_ESTIMATOR_NATIVE_SOURCES = ${FACEMC_ESTIMATOR_NATIVE_SOURCES}")
ENDIF()

SET(SUBPACKAGE_LIB_NAME facemc_estimator_native)

# Create the utilitycore library
ADD_LIBRARY(${SUBPACKAGE_LIB_NAME} ${FACEMC_ESTIMATOR_NATIVE_SOURCES})
TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${TEUCHOS_CORE} utility_dist facemc_core geometry_dagmc)
IF(${FACEMC_ENABLE_MPI})
  TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${MPI_CXX_LIBRARIES})
ENDIF()

INSTALL(TARGETS ${SUBPACKAGE_LIB_NAME}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/lib)
INSTALL(FILES ${FACEMC_ESTIMATOR_NATIVE_HEADERS}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/include
  PERMISSIONS OWNER_READ GROUP_READ WORLD_READ)

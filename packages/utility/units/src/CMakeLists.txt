# Find all headers and sources (no updates to this file necessary!)
FILE(GLOB UTILITY_UNITS_HEADERS *.hpp)
FILE(GLOB UTILITY_UNITS_SOURCES *.cpp *.F90)

IF(CMAKE_VERBOSE_CONFIGURE)
  MESSAGE("UTILITY_UNITS_HEADERS = ${UTILITY_UNITS_HEADERS}")
  MESSAGE("UTILITY_UNITS_SOURCES = ${UTILITY_UNITS_SOURCES}")
ENDIF()

SET(SUBPACKAGE_LIB_NAME utility_units)

# Create the subpackage library
ADD_LIBRARY(${SUBPACKAGE_LIB_NAME} ${UTILITY_UNITS_SOURCES})
TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${TEUCHOS_CORE})

INSTALL(TARGETS ${SUBPACKAGE_LIB_NAME}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/lib)
INSTALL(FILES ${UTILITY_UNITS_HEADERS}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/include
  PERMISSIONS OWNER_READ GROUP_READ WORLD_READ)

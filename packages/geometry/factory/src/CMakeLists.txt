# Find all headers and sources (no updates to this file necessary!)
FILE(GLOB GEOMETRY_FACTORY_HEADERS *.hpp)
FILE(GLOB GEOMETRY_FACTORY_SOURCES *.cpp)

IF(CMAKE_VERBOSE_CONFIGURE)
  MESSAGE("GEOMETRY_FACTORY_HEADERS = ${GEOMETRY_FACTORY_HEADERS}")
  MESSAGE("GEOMETRY_FACTORY_SOURCES = ${GEOMETRY_FACTORY_SOURCES}")
ENDIF()

SET(SUBPACKAGE_LIB_NAME geometry_factory)

# Create the geometry_core library
ADD_LIBRARY(${SUBPACKAGE_LIB_NAME} ${GEOMETRY_FACTORY_SOURCES})
TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} ${TEUCHOS_CORE} ${TEUCHOS_COMM} utility_core)

IF(FRENSIE_ENABLE_ROOT)
  TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} geometry_root)
ENDIF()

IF(FRENSIE_ENABLE_DAGMC)
  TARGET_LINK_LIBRARIES(${SUBPACKAGE_LIB_NAME} geometry_dagmc)
ENDIF()

INSTALL(TARGETS ${SUBPACKAGE_LIB_NAME}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/lib)
INSTALL(FILES ${GEOMETRY_FACTORY_HEADERS}
  DESTINATION ${CMAKE_INSTALL_PREFIX}/include
  PERMISSIONS OWNER_READ GROUP_READ WORLD_READ)
